CLASS lhc_LeaveReq DEFINITION INHERITING FROM cl_abap_behavior_handler.
  PRIVATE SECTION.

    METHODS get_instance_features FOR INSTANCE FEATURES
      IMPORTING keys REQUEST requested_features FOR LeaveReq
      RESULT result.

    METHODS setInitialStatus FOR DETERMINE ON MODIFY
      IMPORTING keys FOR LeaveReq~setInitialStatus.

    METHODS calculateLeaveDays FOR DETERMINE ON MODIFY
      IMPORTING keys FOR LeaveReq~calculateLeaveDays.

    METHODS validateDates FOR VALIDATE ON SAVE
      IMPORTING keys FOR LeaveReq~validateDates.

    METHODS submit FOR MODIFY
      IMPORTING keys FOR ACTION LeaveReq~submit RESULT result.

    METHODS approve FOR MODIFY
      IMPORTING keys FOR ACTION LeaveReq~approve RESULT result.

    METHODS reject FOR MODIFY
      IMPORTING keys FOR ACTION LeaveReq~reject RESULT result.



ENDCLASS.

CLASS lhc_LeaveReq IMPLEMENTATION.

  METHOD get_instance_features.

  READ ENTITIES OF z_i_leavereq IN LOCAL MODE
    ENTITY LeaveReq
    FIELDS ( status )
    WITH CORRESPONDING #( keys )
    RESULT DATA(lt_leave).

  result = VALUE #(
    FOR ls_leave IN lt_leave
    (
      %tky = ls_leave-%tky

      %action-submit =
        COND #(
          WHEN ls_leave-status = 'N'
          THEN if_abap_behv=>fc-o-enabled
          ELSE if_abap_behv=>fc-o-disabled
        )

      %action-approve =
        COND #(
          WHEN ls_leave-status = 'S'
          THEN if_abap_behv=>fc-o-enabled
          ELSE if_abap_behv=>fc-o-disabled
        )

      %action-reject =
        COND #(
          WHEN ls_leave-status = 'S'
          THEN if_abap_behv=>fc-o-enabled
          ELSE if_abap_behv=>fc-o-disabled
        )
    )
  ).

ENDMETHOD.

  METHOD setInitialStatus.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( status )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    DATA lt_update TYPE TABLE FOR UPDATE z_i_leavereq.

    LOOP AT lt_leave INTO DATA(ls_leave).
      IF ls_leave-status IS INITIAL.
        APPEND VALUE #(
          %tky   = ls_leave-%tky
          status = 'N'
        ) TO lt_update.
      ENDIF.
    ENDLOOP.

    IF lt_update IS NOT INITIAL.
      MODIFY ENTITIES OF z_i_leavereq IN LOCAL MODE
        ENTITY LeaveReq
        UPDATE FIELDS ( status )
        WITH lt_update.
    ENDIF.

  ENDMETHOD.



  METHOD calculateLeaveDays.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( date_from date_to leave_days )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    DATA lt_update TYPE TABLE FOR UPDATE z_i_leavereq.

    LOOP AT lt_leave INTO DATA(ls_leave).

      IF ls_leave-date_from IS INITIAL OR ls_leave-date_to IS INITIAL.
        CONTINUE.
      ENDIF.

      IF ls_leave-date_to < ls_leave-date_from.
        CONTINUE.
      ENDIF.

      DATA(lv_days) = ls_leave-date_to - ls_leave-date_from + 1.

      APPEND VALUE #(
        %tky       = ls_leave-%tky
        leave_days = lv_days
      ) TO lt_update.

    ENDLOOP.

    IF lt_update IS NOT INITIAL.
      MODIFY ENTITIES OF z_i_leavereq IN LOCAL MODE
        ENTITY LeaveReq
        UPDATE FIELDS ( leave_days )
        WITH lt_update.
    ENDIF.

  ENDMETHOD.



  METHOD validateDates.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( date_from date_to )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    LOOP AT lt_leave INTO DATA(ls_leave).

      IF ls_leave-date_from IS INITIAL OR ls_leave-date_to IS INITIAL.
        APPEND VALUE #( %tky = ls_leave-%tky ) TO failed-LeaveReq.

      ELSEIF ls_leave-date_to < ls_leave-date_from.
        APPEND VALUE #( %tky = ls_leave-%tky ) TO failed-LeaveReq.

      ENDIF.

    ENDLOOP.

  ENDMETHOD.



  METHOD submit.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( status )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    DATA lt_update TYPE TABLE FOR UPDATE z_i_leavereq.

    LOOP AT lt_leave INTO DATA(ls_leave).

      IF ls_leave-status = 'N'.
        APPEND VALUE #(
          %tky   = ls_leave-%tky
          status = 'S'
        ) TO lt_update.

      ELSE.
        APPEND VALUE #( %tky = ls_leave-%tky ) TO failed-LeaveReq.

        APPEND VALUE #(
          %tky = ls_leave-%tky
          %msg = new_message_with_text(
                   severity = if_abap_behv_message=>severity-error
                   text     = 'Only New requests can be submitted'
                 )
        ) TO reported-LeaveReq.
      ENDIF.

    ENDLOOP.

    IF lt_update IS NOT INITIAL.
      MODIFY ENTITIES OF z_i_leavereq IN LOCAL MODE
        ENTITY LeaveReq
        UPDATE FIELDS ( status )
        WITH lt_update.
    ENDIF.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      ALL FIELDS
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_result).

    result = VALUE #( FOR ls_result IN lt_result
                      ( %tky   = ls_result-%tky
                        %param = ls_result ) ).

  ENDMETHOD.



  METHOD approve.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( status )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    DATA lt_update TYPE TABLE FOR UPDATE z_i_leavereq.

    LOOP AT lt_leave INTO DATA(ls_leave).

      IF ls_leave-status = 'S'.
        APPEND VALUE #(
          %tky   = ls_leave-%tky
          status = 'A'
        ) TO lt_update.

      ELSE.
        APPEND VALUE #( %tky = ls_leave-%tky ) TO failed-LeaveReq.

        APPEND VALUE #(
          %tky = ls_leave-%tky
          %msg = new_message_with_text(
                   severity = if_abap_behv_message=>severity-error
                   text     = 'Only Submitted requests can be approved'
                 )
        ) TO reported-LeaveReq.
      ENDIF.

    ENDLOOP.

    IF lt_update IS NOT INITIAL.
      MODIFY ENTITIES OF z_i_leavereq IN LOCAL MODE
        ENTITY LeaveReq
        UPDATE FIELDS ( status )
        WITH lt_update.
    ENDIF.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      ALL FIELDS
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_result).

    result = VALUE #( FOR ls_result IN lt_result
                      ( %tky   = ls_result-%tky
                        %param = ls_result ) ).

  ENDMETHOD.



  METHOD reject.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      FIELDS ( status )
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_leave).

    DATA lt_update TYPE TABLE FOR UPDATE z_i_leavereq.

    LOOP AT lt_leave INTO DATA(ls_leave).

      IF ls_leave-status = 'S'.
        APPEND VALUE #(
          %tky   = ls_leave-%tky
          status = 'R'
        ) TO lt_update.

      ELSE.
        APPEND VALUE #( %tky = ls_leave-%tky ) TO failed-LeaveReq.

        APPEND VALUE #(
          %tky = ls_leave-%tky
          %msg = new_message_with_text(
                   severity = if_abap_behv_message=>severity-error
                   text     = 'Only Submitted requests can be rejected'
                 )
        ) TO reported-LeaveReq.
      ENDIF.

    ENDLOOP.

    IF lt_update IS NOT INITIAL.
      MODIFY ENTITIES OF z_i_leavereq IN LOCAL MODE
        ENTITY LeaveReq
        UPDATE FIELDS ( status )
        WITH lt_update.
    ENDIF.

    READ ENTITIES OF z_i_leavereq IN LOCAL MODE
      ENTITY LeaveReq
      ALL FIELDS
      WITH CORRESPONDING #( keys )
      RESULT DATA(lt_result).

    result = VALUE #( FOR ls_result IN lt_result
                      ( %tky   = ls_result-%tky
                        %param = ls_result ) ).

  ENDMETHOD.

ENDCLASS.
